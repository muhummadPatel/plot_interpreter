// TEST: LOOP AROUND THE PLOTTABLE SURFACE
//plot_interpreter.ino

#include <grid.h>

Grid grid;
long origin[2];
long pos[2];
void setup() {
    origin[0] = 0;
    origin[1] = 0;

    pos[0] = 13;
    pos[1] = 13;
}

void loop() {
    pos[0] = 0;
    pos[1] = 13;

    grid.moveToPixels(pos);
    delay(500);

    pos[0] = 13;
    pos[1] = 13;

    grid.moveToPixels(pos);
    delay(500);

    pos[0] = 13;
    pos[1] = 0;

    grid.moveToPixels(pos);
    delay(500);

    pos[0] = 0;
    pos[1] = 0;

    grid.moveToPixels(origin);
    delay(500);
}
